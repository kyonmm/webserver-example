=== そのまま動かしてみる
----
cd .\nginx\stable\alpine\
docker build -t kyonmm/nginx .
docker run -p 8081:80 -d kyonmm/nginx
docker ps
wget http://localhost:8081
docker ps -a | peco | % {($_ -split "\s+")[0]} | %{ docker rm -f $_ }
